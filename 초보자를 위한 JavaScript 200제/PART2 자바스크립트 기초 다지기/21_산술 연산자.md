# 산술 연산자
```
10 + 5; // 15
10 - 5; // 10
10 * 5; // 50
10 / 5; // 2
"문자열" + "이어 붙이기"; // 문자열 이어 붙이기
```
```
var x = 10;
x += 5;
x *= 2;
console.log(x); // 30
var y = 10;
y -= 5;
y /= 5;
console.log(y); // 1
```
```
console.log(15 % 4); // 3
console.log(3 ** 3); // 27
console.log(+10); // 10
console.log(-10); // -10
var value = 10;
++value;
--value;
console.log(value); // 10
```
